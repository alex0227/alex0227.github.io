多网卡默认监听eth0，指定DHCP监听eth1网卡
```
[root@linux-node1 ~]# vim /etc/sysconfig/dhcpd
# Command line options here
DHCPDARGS=eth1  # 指定监听网卡
[root@linux-node1 ~]# /etc/init.d/dhcpd restart
```
